Recall: the dot product of two vectors is the sum of the product of the components.
  1. \begin{align*} \mbf{u}\cdot \mbf{v} \amp = \begin{bmatrix} 2 \\ -1 \\ 3\end{bmatrix} \cdot \begin{bmatrix} -1 \\ 1 \\ 1 \end{bmatrix} = 2\cdot (-1) + (-1)\cdot 1 + 3\cdot 1 = -2 -1+3 = 0. \end{align*}
  2. \begin{align*} \mbf{u}\cdot \mbf{v} \amp = \begin{bmatrix} 1 \\ 2 \\ -1\end{bmatrix} \cdot \begin{bmatrix} 1 \\ 2 \\ -1\end{bmatrix} = 1\cdot 1 + 2\cdot 2 + (-1)\cdot (-1) = 1+4+1 = 6. \end{align*}
  3. \begin{align*} \mbf{u}\cdot \mbf{v} \amp = \begin{bmatrix} 1 \\ 1 \\ -3\end{bmatrix} \cdot \begin{bmatrix} 2 \\ -1 \\ 1 \end{bmatrix} = 1\cdot 2 + 1\cdot (-1) + (-3)\cdot 1 = 2-1-3=-2 \end{align*}
  4. \begin{align*} \mbf{u}\cdot \mbf{v} \amp = \begin{bmatrix} 3 \\ -1 \\ 5\end{bmatrix} \cdot \begin{bmatrix} 6 \\ -7 \\ -5 \end{bmatrix} = 3\cdot 6 + (-1)\cdot (-7)+5\cdot(-5)\\ \amp = 18 + 7 -25 = 0. \end{align*}
  5. \begin{align*} \mbf{u}\cdot \mbf{v} \amp = \begin{bmatrix} x \\ y \\ z\end{bmatrix} \cdot \begin{bmatrix} a \\ b \\ c \end{bmatrix} = x\cdot a + y\cdot b + z\cdot c. \end{align*}
  6. \begin{align*} \mbf{u}\cdot \mbf{v} \amp = \begin{bmatrix} x \\ y \\ z\end{bmatrix} \cdot \begin{bmatrix} 0 \\ 0 \\ 0 \end{bmatrix} = x\cdot 0 + y\cdot 0 + z\cdot 0 = 0 + 0 + 0 = 0. \end{align*}
