\begin{align} (\mathbf{u} + \mathbf{v})\cdot(\mathbf{u} - \mathbf{v}) \amp = (\mathbf{u} + \mathbf{v})\cdot\mathbf{u} - (\mathbf{u} + \mathbf{v})\cdot\mathbf{v} \amp\amp (\text{Distributivity})\notag\\ \amp = (\mathbf{u}\cdot\mathbf{u} + \mathbf{v}\cdot\mathbf{u}) - (\mathbf{u}\cdot\mathbf{v} + \mathbf{v}\cdot\mathbf{v}) \amp\amp (\text{Distributivity})\notag\\ \amp = \norm{\mbf{u}}^2 + \mathbf{v}\cdot\mathbf{u} - \mathbf{u}\cdot\mathbf{v} - \norm{\mbf{v}}^2 \amp\amp (\text{definition of } \norm{\cdot})\notag\\ \amp = \norm{\mbf{u}}^2 + \mathbf{u}\cdot \mathbf{v} - \mathbf{u}\cdot\mathbf{v} - \norm{\mbf{v}}^2 \amp\amp (\text{Commutativity})\notag\\ \amp = \norm{\mbf{u}}^2 - \norm{\mbf{v}}^2 .\tag{1.2.2} \end{align}
