For a \(3\times3 \) matrix, expanding along the \(1\)st column,
\begin{equation*} cof(A)_{11} = (-1)^{1+1}\begin{vmatrix} 0 \amp 1 \\ 1 \amp 1 \end{vmatrix} =-1 \end{equation*}
\begin{equation*} cof(A)_{21} = (-1)^{2+1}\begin{vmatrix} 2 \amp 1 \\ 1 \amp 1 \end{vmatrix}=(-1)(1)=-1 \end{equation*}
\begin{equation*} cof(A)_{31} = (-1)^{3+1}\begin{vmatrix} 1 \amp 2 \\ 1 \amp 0 \end{vmatrix}=-2 \end{equation*}
\begin{equation*} \det(A) = a_{11}cof(A)_{11}+a_{21}cof(A)_{21}+a_{31}cof(A)_{31}=(1)(-1) + (1)(-1)+ (2)(-2)=-6 \end{equation*}
For a \(3\times3 \) matrix, expanding along the \(3\)rd row,
\begin{equation*} cof(A)_{31} = (-1)^{3+1}\begin{vmatrix} 2 \amp 1 \\ 0 \amp 1 \end{vmatrix} =-2 \end{equation*}
\begin{equation*} cof(A)_{32} = (-1)^{3+2}\begin{vmatrix} 1 \amp 1 \\ 1 \amp 1 \end{vmatrix}=0 \end{equation*}
\begin{equation*} cof(A)_{33} = (-1)^{3+3}\begin{vmatrix} 1 \amp 2 \\ 1 \amp 0 \end{vmatrix}=-2 \end{equation*}
\begin{equation*} \det(A) = a_{12}cof(A)_{12}+a_{22}cof(A)_{22}+a_{32}cof(A)_{32}=(2)(-2) + (1)(0)+ (1)(-2)=-6 \end{equation*}
